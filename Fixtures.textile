<pre>
<code>
Before do
  Fixtures.reset_cache
  fixtures = Dir[File.join(RAILS_ROOT, 'spec/fixtures', '*.yml')].map{ |f| File.basename(f, '.yml') }
  Fixtures.create_fixtures('spec/fixtures', fixtures)
end
